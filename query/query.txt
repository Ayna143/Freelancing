create table freelancer (
    account_id INT AUTO_INCREMENT PRIMARY KEY,
    firstname VARCHAR (100) NOT NULL,
    lastname VARCHAR (100) NOT NULL,
    email VARCHAR (100) NOT NULL UNIQUE,
    password VARCHAR(100) NOT NULL,
    address VARCHAR(100) NOT NULL,
    profile_pic VARCHAR(255),
    about_id INT,
    FOREIGN KEY(about_id) REFERENCES about(about_id)
);

CREATE TABLE work (
    work_id INT AUTO_INCREMENT PRIMARY KEY,
    freelancer_id INT NOT NULL,
    picture VARCHAR(255) NOT NULL,
    title VARCHAR(100) NOT NULL,
    description VARCHAR(100),
    category VARCHAR(100) NOT NULL,
    date TIMESTAMP CURRENT_TIMESTAMP,
    FOREIGN KEY (freelancer_id) REFERENCES freelancer(account_id)
 );

create table job(
    job_id INT AUTO_INCREMENT PRIMARY KEY,
    ClientId INT NOT NULL,
    FullName VARCHAR(100) NOT NULL,
    Email VARCHAR(100) NOT NULL,
    Project_Category VARCHAR(100) NOT NULL,
    Description TEXT NOT NULL,
    start_date DATE,
    end_date DATE,
    Budget DECIMAL(10, 2),
    Project_Name VARCHAR(100),
    Date_created TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    Location VARCHAR(100) NOT NULL,
    picture VARCHAR(255),
    FOREIGN KEY (ClientId) REFERENCES client(client_id)
)

create table client (
    client_id INT AUTO_INCREMENT PRIMARY KEY,
    firstname VARCHAR (100) NOT NULL,
    lastname VARCHAR (100) NOT NULL,
    email VARCHAR (100) NOT NULL UNIQUE,
    password VARCHAR(100) NOT NULL,
    address VARCHAR(100) NOT NULL,
    profile_pic VARCHAR(255),
    about_id INT,
    FOREIGN KEY(about_id) REFERENCES about(about_id)
);

create table about(
    about_id INT AUTO_INCREMENT PRIMARY KEY,
    contact INT(11) NOT NULL,
    birthday DATE NOT NULL,
    skills TEXT NOT NULL,
    history TEXT NOT NULL,
    socials TEXT NOT NULL
);

ALTER TABLE client
ADD CONSTRAINT AboutId
FOREIGN KEY (about_id) REFERENCES about(about_id);

ALTER TABLE freelancer
ADD CONSTRAINT About_Id
FOREIGN KEY (about_id) REFERENCES about(about_id);
